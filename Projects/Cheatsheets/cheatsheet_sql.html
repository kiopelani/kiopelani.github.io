SQL CHEATSHEET
Create a New Table
CREATE TABLE
CREATE TABLE table_name (
    column_name1 type,
    column_name2 type
);
Add Data Records to a Table
INSERT INTO
INSERT INTO table_name
    (column_name1, column_name2)
VALUES
    ('Value 1', 'Value 2');
Select Data from Database
SELECT...FROM
To select data from a single column in a table:
SELECT column_name1
FROM table_name;

To select data from multiple columns in a table:
SELECT column_name1, column_name2
FROM table_name;

To select all data in a table:
SELECT *
FROM table_name;
Select Data By a Specified Condition
WHERE Clause
SELECT column_name1
FROM table_name
WHERE column_name operator some_value;

Example:
SELECT pet_name
FROM dogs
WHERE breed = 'Pug';

Combine/Select Data From Multiple Tables
INNER JOIN (or JOIN)
SELECT Table1.column1 FROM Table1
JOIN Table2 ON Table1.column1 = Table2.column1;
